еҹәжң¬жҰӮеҝө

йёҝи’ҷеҶ…ж ёдёӯзҡ„зәҝзЁӢйҮҮз”ЁжҠўеҚ ејҸи°ғеәҰжңәеҲ¶пјҢеҗҢж—¶ж”ҜжҢҒж—¶й—ҙзүҮиҪ®иҪ¬и°ғеәҰе’ҢFIFOи°ғеәҰж–№ејҸгҖӮ

йёҝи’ҷеҶ…ж ёзҡ„зәҝзЁӢдёҖе…ұжңү32дёӘдјҳе…Ҳзә§(0-31)пјҢжңҖй«ҳдјҳе…Ҳзә§дёә0пјҢжңҖдҪҺдјҳе…Ҳзә§дёә31гҖӮ

{
    UINT32 index;
    UINT32 ret;
    UINT32 size;

    g_taskMaxNum = LOSCFG_BASE_CORE_TSK_LIMIT;//д»»еҠЎжұ дёӯжңҖеӨҡй»ҳи®Ө128дёӘ,еҸҜи°“й“Ғжү“зҡ„д»»еҠЎжұ жөҒж°ҙзҡ„зәҝзЁӢ

    /*
     * This memory is resident memory and is used to save the system resources
     * of task control block and will not be freed.
     */
    g_taskCBArray = (LosTaskCB *)LOS_MemAlloc(m_aucSysMem0, size);//д»»еҠЎжұ  еёёй©»еҶ…еӯҳ,дёҚиў«йҮҠж”ҫ
    if (g_taskCBArray == NULL) {
        return LOS_ERRNO_TSK_NO_MEMORY;
    }
    (VOID)memset_s(g_taskCBArray, size, 0, size);

    LOS_ListInit(&g_losFreeTask);//з©әй—Ід»»еҠЎй“ҫиЎЁ
    LOS_ListInit(&g_taskRecyleList);//йңҖеӣһж”¶д»»еҠЎй“ҫиЎЁ
    for (index = 0; index < g_taskMaxNum; index++) {
        g_taskCBArray[index].taskStatus = OS_TASK_STATUS_UNUSED;
        g_taskCBArray[index].taskID = index;//д»»еҠЎIDжңҖеӨ§й»ҳи®Ө127
        LOS_ListTailInsert(&g_losFreeTask, &g_taskCBArray[index].pendList);//йғҪжҸ’е…Ҙз©әй—Ід»»еҠЎеҲ—иЎЁ 
    }//жіЁж„Ҹ:иҝҷйҮҢжҢӮзҡ„жҳҜpendListиҠӮзӮ№,жүҖд»ҘеҸ–TCBиҰҒйҖҡиҝҮ OS_TCB_FROM_PENDLIST еҸ–.

    ret = OsPriQueueInit();//еҲӣе»ә32дёӘд»»еҠЎдјҳе…Ҳзә§йҳҹеҲ—пјҢеҚі32дёӘеҸҢеҗ‘еҫӘзҺҜй“ҫиЎЁ
    if (ret != LOS_OK) {
        return LOS_ERRNO_TSK_NO_MEMORY;
    }

    for (index = 0; index < LOSCFG_KERNEL_CORE_NUM; index++) {
        ret = OsSortLinkInit(&g_percpu[index].taskSortLink);//жҜҸдёӘCPUеҶ…ж ёйғҪжңүдёҖдёӘжү§иЎҢд»»еҠЎй“ҫиЎЁ
        if (ret != LOS_OK) {
            return LOS_ERRNO_TSK_NO_MEMORY;
        }
    }
    return LOS_OK;
}

{
   
    UINT32 index = 1;
    TaskContext *taskContext = NULL;

    if (initFlag == TRUE) {
   
        OsStackInit(topStack, stackSize);
    }
    taskContext = (TaskContext *)(((UINTPTR)topStack + stackSize) - sizeof(TaskContext));//жіЁж„ҸзңӢдёҠдёӢж–Үе°Ҷеӯҳж”ҫеңЁж Ҳзҡ„еә•йғЁ

    /* initialize the task context */
#ifdef LOSCFG_GDB
    taskContext->PC = (UINTPTR)OsTaskEntrySetupLoopFrame;
#else
    taskContext->PC = (UINTPTR)OsTaskEntry;//зЁӢеәҸи®Ўж•°еҷЁ,CPUйҰ–ж¬Ўжү§иЎҢtaskж—¶и·‘зҡ„з¬¬дёҖжқЎжҢҮд»ӨдҪҚзҪ®
#endif
    taskContext->LR = (UINTPTR)OsTaskExit;  /* LR should be kept, to distinguish it's THUMB or ARM instruction */
    taskContext->resved = 0x0;
    taskContext->R[0] = taskID;             /* R0 */
    taskContext->R[index++] = 0x01010101;   /* R1, 0x01010101 : reg initialed magic word */
    for (; index < GEN_REGS_NUM; index++) {
   //R2 - R12зҡ„еҲқе§ӢеҢ–еҫҲжңүж„ҸжҖқ,дёәд»Җд№ҲиҰҒиҝҷд№ҲеҒҡпјҹ
        taskContext->R[index] = taskContext->R[index - 1] + taskContext->R[1]; /* R2 - R12 */
    }

#ifdef LOSCFG_INTERWORK_THUMB // 16дҪҚжЁЎејҸ
    taskContext->regPSR = PSR_MODE_SVC_THUMB; /* CPSR (Enable IRQ and FIQ interrupts, THUMNB-mode) */
#else
    taskContext->regPSR = PSR_MODE_SVC_ARM;   /* CPSR (Enable IRQ and FIQ interrupts, ARM-mode) */
#endif

#if !defined(LOSCFG_ARCH_FPU_DISABLE)
    /* 0xAAA0000000000000LL : float reg initialed magic word */
    for (index = 0; index < FP_REGS_NUM; index++) {
   
        taskContext->D[index] = 0xAAA0000000000000LL + index; /* D0 - D31 */
    }
    taskContext->regFPSCR = 0;
    taskContext->regFPEXC = FP_EN;
#endif

    return (VOID *)taskContext;
}

//еҲӣе»әTask
LITE_OS_SEC_TEXT_INIT UINT32 LOS_TaskCreate(UINT32 *taskID, TSK_INIT_PARAM_S *initParam)
{
    UINT32 ret;
    UINT32 intSave;
    LosTaskCB *taskCB = NULL;

    if (initParam == NULL) {
        return LOS_ERRNO_TSK_PTR_NULL;
    }

    if (OS_INT_ACTIVE) {
        return LOS_ERRNO_TSK_YIELD_IN_INT;
    }

    if (initParam->uwResved & OS_TASK_FLAG_IDLEFLAG) {//OS_TASK_FLAG_IDLEFLAG жҳҜеұһдәҺеҶ…ж ё idleиҝӣзЁӢдё“з”Ёзҡ„
        initParam->processID = OsGetIdleProcessID();//иҺ·еҸ–з©әй—ІиҝӣзЁӢ
    } else if (OsProcessIsUserMode(OsCurrProcessGet())) {//еҪ“еүҚиҝӣзЁӢжҳҜеҗҰдёәз”ЁжҲ·жЁЎејҸ
        initParam->processID = OsGetKernelInitProcessID();//дёҚжҳҜе°ұеҸ–"Kernel"иҝӣзЁӢ
    } else {
        initParam->processID = OsCurrProcessGet()->processID;//иҺ·еҸ–еҪ“еүҚиҝӣзЁӢ IDиөӢеҖј
    }
    initParam->uwResved &= ~OS_TASK_FLAG_IDLEFLAG;//дёҚиғҪжҳҜ OS_TASK_FLAG_IDLEFLAG
    initParam->uwResved &= ~OS_TASK_FLAG_PTHREAD_JOIN;//дёҚиғҪжҳҜ OS_TASK_FLAG_PTHREAD_JOIN
    if (initParam->uwResved & LOS_TASK_STATUS_DETACHED) {//жҳҜеҗҰи®ҫзҪ®дәҶиҮӘеҠЁеҲ йҷӨ
        initParam->uwResved = OS_TASK_FLAG_DETACHED;//иҮӘеҠЁеҲ йҷӨ,жіЁж„ҸиҝҷйҮҢжҳҜ = ,д№ҹе°ұжҳҜиҜҙеҸӘжңү OS_TASK_FLAG_DETACHED дёҖдёӘж ҮзӯҫдәҶ
    }

    ret = LOS_TaskCreateOnly(taskID, initParam);//еҲӣе»әдёҖдёӘд»»еҠЎ,иҝҷжҳҜд»»еҠЎеҲӣе»әзҡ„е®һдҪ“,еүҚйқўйғҪеҸӘжҳҜеүҚжңҹеҮҶеӨҮе·ҘдҪң
    if (ret != LOS_OK) {
        return ret;
    }
    taskCB = OS_TCB_FROM_TID(*taskID);//йҖҡиҝҮIDжӢҝеҲ°taskе®һдҪ“

    SCHEDULER_LOCK(intSave);
    taskCB->taskStatus &= ~OS_TASK_STATUS_INIT;//д»»еҠЎдёҚеҶҚжҳҜеҲқе§ӢеҢ–
    OS_TASK_SCHED_QUEUE_ENQUEUE(taskCB, 0);//иҝӣе…Ҙи°ғеәҰе°ұз»ӘйҳҹеҲ—,ж–°д»»еҠЎжҳҜзӣҙжҺҘиҝӣе…Ҙе°ұз»ӘйҳҹеҲ—зҡ„
    SCHEDULER_UNLOCK(intSave);

    /* in case created task not running on this core,
       schedule or not depends on other schedulers status. */
    LOS_MpSchedule(OS_MP_CPU_ALL);//еҰӮжһңеҲӣе»әзҡ„д»»еҠЎжІЎжңүеңЁиҝҷдёӘж ёеҝғдёҠиҝҗиЎҢпјҢжҳҜеҗҰи°ғеәҰеҸ–еҶідәҺе…¶д»–и°ғеәҰзЁӢеәҸзҡ„зҠ¶жҖҒгҖӮ
    if (OS_SCHEDULER_ACTIVE) {//еҪ“еүҚCPUж ёеӨ„дәҺеҸҜи°ғеәҰзҠ¶жҖҒ
        LOS_Schedule();//еҸ‘иө·и°ғеәҰ
    }

    return LOS_OK;
}
